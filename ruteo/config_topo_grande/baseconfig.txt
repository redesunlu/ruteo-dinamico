!The %h in the following line gets converted to your GNS3 router name
hostname %h
!Prevent typing mistakes firing off DNS lookups
no ip domain-lookup
line con 0
!Prevent the console timing out - AND this also helps reduce your CPU utilisation
exec-timeout 0 0
!You don't want console messages appearing in the middle of your typing
logging synchronous
! and if you never want to have to type "enable" again, add:
privilege level 15
